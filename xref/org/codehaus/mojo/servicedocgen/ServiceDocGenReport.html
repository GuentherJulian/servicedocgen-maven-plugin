51  /**
52   * {@link AbstractMojo Maven Plugin} to automatically generate documentation for services of the current project.
53   * <ol>
54   * <li>Scans the current projects source code for (JAX-RS annotated) services that match the RegEx configured by
55   * <code>classnameRegex</code>.</li>
56   * <li>Analyzes the services from source-code (extract JavaDoc, etc.) and byte-code (resolve generic parameters, etc.)
57   * and create intermediate meta-data as {@link ServicesDescriptor}.</li>
58   * <li>Generates documentation from the collected meta-data (by default as HTML from a velocity template shipped with
59   * this plugin but can be overridden via configuration parameters).</li>
60   * </ol>
61   * scan and analyze services from the current project via source-code and byte-code analysis. Creates
62   *
63   * @author hohwille
64   */
65  @Mojo( name = "generate", defaultPhase = LifecyclePhase.PREPARE_PACKAGE, requiresProject = true, requiresDirectInvocation = false, executionStrategy = "once-per-session", requiresDependencyCollection = ResolutionScope.COMPILE_PLUS_RUNTIME )
66  public class ServiceDocGenReport
67      extends AbstractMavenReport
68  {

155     /**
156      * {@inheritDoc}
157      */
158     @Override
159     protected void executeReport( Locale locale )
160         throws MavenReportException
161     {
162         this.classnamePattern = Pattern.compile( this.classnameRegex );
163 
164         JavaProjectBuilder builder = new JavaProjectBuilder();
165         if ( !Util.isEmpty( this.sourceEncoding ) )
166         {
167             builder.setEncoding( this.sourceEncoding );
168         }
169 
170         try
171         {
172             List<JavaClass> serviceClasses = scanServices( builder );
173             if ( serviceClasses.isEmpty() )
174             {
175                 getLog().info( "No services found - omitting service documentation generation." );
176                 return;
177             }
178             Analyzer analyzer = new Analyzer( getLog(), this.project, getProjectClassloader(), builder, this.descriptor,
179                                               this.introspectFields );
180             ServicesDescriptor services = analyzer.createServicesDescriptor( serviceClasses );
181 
182             getLog().info( "Generating output..." );
183             ServicesGenerator generator =
184                 new VelocityServicesGenerator( Util.appendPath( this.templatePath, this.templateName ) );
185 
186             File reportDirectory = new File( this.outputDirectory, this.reportFolder );
187             if ( !reportDirectory.isDirectory() )
188             {
189                 boolean ok = reportDirectory.mkdirs();
190                 if ( !ok )
191                 {
192                     throw new MojoExecutionException( "Could not create directory " + reportDirectory );
193                 }
194             }
195             generator.generate( services, reportDirectory );
196         }
197         catch ( MavenReportException e )
198         {
199             throw e;
200         }
201         catch ( Exception e )
202         {
203             throw new MavenReportException( "Unexpected Error!", e );
204         }
205     }
206 
207     private List<JavaClass> scanServices( JavaProjectBuilder builder )
208         throws IOException
209     {
210         List<JavaClass> serviceClasses = new ArrayList<JavaClass>();
211         for ( String sourceDir : this.project.getCompileSourceRoots() )
212         {
213             File sourceFolder = new File( sourceDir );
214             if ( sourceFolder.isDirectory() )
215             {
216                 builder.addSourceFolder( sourceFolder );
217                 if ( this.serviceClassName == null )
218                 {
219                     scanJavaFilesRecursive( sourceFolder, builder, serviceClasses );
220                 }
221             }
222         }
223         if ( this.serviceClassName != null )
224         {
225             JavaClass type = builder.getClassByName( this.serviceClassName );
226             boolean isService = isServiceClass( type );
227             if ( isService )
228             {
229                 serviceClasses.add( type );
230             }
231         }
232         return serviceClasses;
233     }
234 
235     private void scanJavaFilesRecursive( File sourceDir, JavaProjectBuilder builder, List<JavaClass> serviceClasses )
236         throws IOException
237     {
238         File[] children = sourceDir.listFiles();
239         if ( children == null )
240         {
241             getLog().debug( "Directory does not exist: " + sourceDir );
242             return;
243         }
244         for ( File file : children )
245         {
246             if ( file.isDirectory() )
247             {
248                 scanJavaFilesRecursive( file, builder, serviceClasses );
249             }
250             else if ( file.getName().endsWith( ".java" ) )
251             {
252                 try
253                 {
254                     JavaSource source = builder.addSource( file );
255                     for ( JavaClass type : source.getClasses() )
256                     {
257                         boolean isService = isServiceClass( type );
258                         if ( isService )
259                         {
260                             serviceClasses.add( type );
261                         }
262                     }
263                 }
264                 catch ( Exception e )
265                 {
266                     getLog().debug( "Error parsing file: " + file, e );
267                 }
268             }
269         }
270 
271     }
272 
273     private boolean isServiceClass( JavaClass type )
274     {
275         if ( this.classnamePattern.matcher( type.getName() ).matches() )
276         {
277             getLog().debug( "Class matches: " + type.getName() );
278             for ( JavaAnnotation annotation : type.getAnnotations() )
279             {
280                 if ( Path.class.getName().equals( annotation.getType().getFullyQualifiedName() ) )
281                 {
282                     getLog().info( "Found service: " + type.getName() );
283                     return true;
284                 }
285             }
286         }
287         return false;
288     }

290     private ClassLoader getProjectClassloader()
291         throws MojoExecutionException
292     {
293         if ( this.projectClassloader == null )
294         {
295             this.projectClassloader = new URLClassLoader( buildClasspathUrls(), this.getClass().getClassLoader() );
296         }
297         return this.projectClassloader;
298     }
299 
300     private URL[] buildClasspathUrls()
301         throws MojoExecutionException
302     {
303         List<URL> urls = new ArrayList<URL>( this.runtimeClasspathElements.size() );
304         for ( String element : this.runtimeClasspathElements )
305         {
306             try
307             {
308                 URL url = new File( element ).toURI().toURL();
309                 urls.add( url );
310                 getLog().debug( "Adding to classloader: " + url.toString() );
311             }
312             catch ( MalformedURLException e )
313             {
314                 throw new MojoExecutionException( "Unable to access project dependency: " + element, e );
315             }
316         }
317         return urls.toArray( new URL[urls.size()] );
318     }
